1. Database Create
CREATE DATABASE project;

2. users table
CREATE TABLE users(
    id int PRIMARY KEY AUTO_INCREMENT,
    name varchar(70),
    email varchar(70),
    password varchar(100),
    role varchar(60),
    dept_id int,
    FOREIGN KEY(dept_id) REFERENCES department(id),
    status boolean default false
    );

3. deparment table
CREATE TABLE department(
    id int PRIMARY KEY AUTO_INCREMENT,
    name varchar(70),
    short_name varchar(30),
    established date
    );
4. course table
CREATE TABLE course(
    id int PRIMARY KEY AUTO_INCREMENT,
    course_title varchar(100),
    course_code varchar(20),
    credit float
    );
5. session table
CREATE TABLE session(
    id int PRIMARY KEY AUTO_INCREMENT,
    session varchar(70)
    );
6. batch table
CREATE TABLE batch(
    id int PRIMARY KEY AUTO_INCREMENT,
    batch int,
    dept_id int,
    FOREIGN KEY(dept_id) REFERENCES department(id)
    );
7. section table
CREATE TABLE section(
    id int PRIMARY KEY AUTO_INCREMENT,
    section varchar(20),
    session_id int,
    dept_id int,
    semester varchar(20),
    FOREIGN KEY (session_id) REFERENCES session(id),
    FOREIGN KEY (dept_id) REFERENCES department(id)
    );
8. offer_course table
CREATE TABLE offer_course(
    id int PRIMARY KEY AUTO_INCREMENT,
    semester varchar(20),
    dept_id int,
    session_id int,
    course_id int,
    FOREIGN KEY (dept_id) REFERENCES department(id),
    FOREIGN KEY (session_id) REFERENCES session(id),
    FOREIGN KEY (course_id) REFERENCES course(id)
    );
9. assign_teacher table
CREATE TABLE assign_teacher(
    id int PRIMARY KEY AUTO_INCREMENT,
    teacher_id int,
    dept_id int,
    course int,
    session_id int,
    semester varchar(30),
    section varchar(20),  
    FOREIGN KEY (teacher_id) REFERENCES users(id),
    FOREIGN KEY (dept_id) REFERENCES department(id),
    FOREIGN KEY (session_id) REFERENCES session(id),
    FOREIGN KEY (course) REFERENCES course(id)
    );
10. admin table
CREATE TABLE admin(
    id int PRIMARY KEY AUTO_INCREMENT,
    name varchar(90),
    email varchar(80),
    password varchar(200),
    dept_id int,
    role varchar(70)
    );
 11. project_idea table 
 CREATE TABLE project_idea(
    id int PRIMARY KEY AUTO_INCREMENT,
    gorup_number int,
    idea varchar(2000),
    course_id int,
    session_id int,
    user_id int,
    FOREIGN KEY(session_id) REFERENCES session(id),
    FOREIGN KEY(course_id) REFERENCES course(id),
     FOREIGN KEY(user_id) REFERENCES users(id)
    );  
12. number_distribution table
CREATE TABLE number_distribution(
    id int PRIMARY KEY AUTO_INCREMENT,
    student_id int,
    course_id int,
    session_id int,
    ct int,
    mid int,
    assignment int,
    attendance int,
    final int,
    FOREIGN KEY(student_id) REFERENCES users(id),
    FOREIGN KEY(course_id) REFERENCES course(id),
    FOREIGN KEY(session_id) REFERENCES session(id)
    );
13. enroll table
CREATE TABLE enroll(
    id int PRIMARY KEY AUTO_INCREMENT,
    course_id int,
    section_id int,
    dept_id int,
    semester varchar(40),
    session_id int,
    types varchar(70),
    user_id int,
    FOREIGN KEY(course_id) REFERENCES course(id),
    FOREIGN KEY(section_id) REFERENCES section(id),
    FOREIGN KEY(session_id) REFERENCES session(id),
    FOREIGN KEY(user_id) REFERENCES users(id)
    );
